    <script>
        // 配置信息
        const API_CONFIG = {
            appkey: 'YRsdNKlaJJU5YibO0OGbYzKWwaklOKSn',
            appid: '2035707181828373248',
            apiUrl: 'https://yuanqi.tencent.com/openapi/v1/agent/chat/completions',
            stream: true
        };

        // 关键修复1：初始化时对话历史为空数组
        // 不要包含任何初始消息，确保第一条消息是user
        let conversationHistory = [];

        // 自动调整输入框高度
        function autoResize(textarea) {
            textarea.style.height = 'auto';
            textarea.style.height = Math.min(textarea.scrollHeight, 100) + 'px';
        }

        // 设置示例问题
        function setExample(text) {
            const input = document.getElementById('userInput');
            input.value = text;
            input.focus();
            autoResize(input);
        }

        // 显示思考状态
        function showThinking() {
            const chatContainer = document.getElementById('chatContainer');
            const thinkingDiv = document.createElement('div');
            thinkingDiv.className = 'thinking';
            thinkingDiv.id = 'thinkingIndicator';
            
            const avatar = document.createElement('div');
            avatar.className = 'avatar assistant-avatar';
            avatar.textContent = '元';
            
            const dots = document.createElement('div');
            dots.className = 'thinking-dots';
            
            for (let i = 0; i < 3; i++) {
                const dot = document.createElement('div');
                dot.className = 'thinking-dot';
                dots.appendChild(dot);
            }
            
            thinkingDiv.appendChild(avatar);
            thinkingDiv.appendChild(dots);
            chatContainer.appendChild(thinkingDiv);
            scrollToBottom();
        }

        // 移除思考状态
        function removeThinking() {
            const indicator = document.getElementById('thinkingIndicator');
            if (indicator) indicator.remove();
        }

        // 添加消息到UI
        function addMessageToUI(role, content) {
            const chatContainer = document.getElementById('chatContainer');
            const messageDiv = document.createElement('div');
            messageDiv.className = `message ${role}-message`;
            
            const avatar = document.createElement('div');
            avatar.className = `avatar ${role}-avatar`;
            avatar.textContent = role === 'user' ? '我' : '元';
            
            const contentDiv = document.createElement('div');
            contentDiv.className = 'message-content';
            contentDiv.textContent = content;
            
            messageDiv.appendChild(avatar);
            messageDiv.appendChild(contentDiv);
            chatContainer.appendChild(messageDiv);
            
            scrollToBottom();
            return { messageDiv, contentDiv };
        }

        // 添加错误消息
        function addErrorMessage(message) {
            const chatContainer = document.getElementById('chatContainer');
            const errorDiv = document.createElement('div');
            errorDiv.className = 'error';
            errorDiv.textContent = message;
            chatContainer.appendChild(errorDiv);
            scrollToBottom();
        }

        // 关键修复2：验证messages数组格式
        function validateMessages(messages) {
            if (messages.length === 0) {
                return { valid: true };
            }
            
            // 检查第一条消息必须是user
            if (messages[0].role !== 'user') {
                return { 
                    valid: false, 
                    reason: "第一条消息的角色必须是'user'，当前是'" + messages[0].role + "'" 
                };
            }
            
            // 检查角色交替
            for (let i = 1; i < messages.length; i++) {
                if (messages[i].role === messages[i-1].role) {
                    return { 
                        valid: false, 
                        reason: `第${i}条消息（角色：${messages[i].role}）与第${i-1}条消息角色相同，违反交替规则` 
                    };
                }
            }
            
            return { valid: true };
        }

        // 关键修复3：修复的发送消息函数
        async function sendMessage() {
            const userInput = document.getElementById('userInput');
            const sendButton = document.getElementById('sendButton');
            const message = userInput.value.trim();
            
            if (!message) return;
            
            // 禁用输入和按钮
            userInput.disabled = true;
            sendButton.disabled = true;
            document.getElementById('apiStatus').textContent = '请求中...';
            
            try {
                // 添加用户消息到UI
                addMessageToUI('user', message);
                
                // 创建当前用户消息（符合API格式）
                const currentUserMessage = {
                    role: 'user',
                    content: [
                        {
                            type: 'text',
                            text: message
                        }
                    ]
                };
                
                // 构建完整的消息数组用于API请求
                const messagesForAPI = [...conversationHistory, currentUserMessage];
                
                // 验证消息格式
                const validation = validateMessages(messagesForAPI);
                if (!validation.valid) {
                    throw new Error(`消息格式验证失败: ${validation.reason}`);
                }
                
                console.log('发送的消息数组:', JSON.stringify(messagesForAPI, null, 2));
                
                // 清空输入框
                userInput.value = '';
                userInput.style.height = 'auto';
                
                // 显示思考状态
                showThinking();
                
                // 构建请求体
                const requestBody = {
                    assistant_id: API_CONFIG.appid,
                    user_id: 'user_' + Date.now(), // 使用唯一用户ID
                    stream: API_CONFIG.stream,
                    messages: messagesForAPI
                };
                
                // 发送请求
                const response = await fetch(API_CONFIG.apiUrl, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': `Bearer ${API_CONFIG.appkey}`
                    },
                    body: JSON.stringify(requestBody)
                });
                
                if (!response.ok) {
                    const errorText = await response.text();
                    throw new Error(`HTTP ${response.status}: ${errorText}`);
                }
                
                removeThinking();
                
                if (API_CONFIG.stream) {
                    await handleStreamResponse(response, currentUserMessage);
                } else {
                    await handleNonStreamResponse(response, currentUserMessage);
                }
                
                document.getElementById('apiStatus').textContent = '就绪';
                
            } catch (error) {
                removeThinking();
                document.getElementById('apiStatus').textContent = '错误';
                addErrorMessage(`请求失败: ${error.message}`);
                console.error('API调用错误:', error);
            } finally {
                userInput.disabled = false;
                sendButton.disabled = false;
                userInput.focus();
            }
        }

        // 关键修复4：修复的流式响应处理
        async function handleStreamResponse(response, currentUserMessage) {
            const reader = response.body.getReader();
            const decoder = new TextDecoder('utf-8');
            let buffer = '';
            let assistantMessage = '';
            let messageElement = null;
            let contentDiv = null;
            
            // 先创建消息容器
            const { messageDiv, contentDiv: msgContentDiv } = addMessageToUI('assistant', '');
            messageElement = messageDiv;
            contentDiv = msgContentDiv;
            
            try {
                while (true) {
                    const { done, value } = await reader.read();
                    if (done) break;
                    
                    buffer += decoder.decode(value, { stream: true });
                    
                    // 按行分割处理
                    const lines = buffer.split('\n');
                    buffer = lines.pop(); // 保存未完成的行
                    
                    for (const line of lines) {
                        if (line.trim() === '') continue;
                        
                        if (line.startsWith('data: ')) {
                            const dataStr = line.substring(6);
                            
                            if (dataStr === '[DONE]') {
                                // 流式响应结束
                                break;
                            }
                            
                            try {
                                const data = JSON.parse(dataStr);
                                
                                if (data.choices && data.choices[0]) {
                                    const choice = data.choices[0];
                                    
                                    if (choice.delta) {
                                        const delta = choice.delta;
                                        
                                        if (delta.content) {
                                            assistantMessage += delta.content;
                                            contentDiv.textContent = assistantMessage;
                                            scrollToBottom();
                                        }
                                    }
                                    
                                    if (choice.finish_reason === 'stop') {
                                        // 正常结束
                                        break;
                                    }
                                }
                            } catch (e) {
                                console.warn('解析流数据失败:', e, '数据:', dataStr);
                            }
                        }
                    }
                }
            } finally {
                // 关键修复5：只有在有实际回复时才更新对话历史
                if (assistantMessage.trim()) {
                    // 添加用户消息到对话历史
                    conversationHistory.push(currentUserMessage);
                    
                    // 添加助手回复到对话历史
                    const assistantMessageObj = {
                        role: 'assistant',
                        content: [
                            {
                                type: 'text',
                                text: assistantMessage
                            }
                        ]
                    };
                    conversationHistory.push(assistantMessageObj);
                }
            }
        }

        // 关键修复6：修复的非流式响应处理
        async function handleNonStreamResponse(response, currentUserMessage) {
            const data = await response.json();
            
            if (data.choices && data.choices[0] && data.choices[0].message) {
                const assistantMessage = data.choices[0].message.content;
                
                // 添加助手消息到UI
                addMessageToUI('assistant', assistantMessage);
                
                // 更新对话历史
                conversationHistory.push(currentUserMessage);
                
                const assistantMessageObj = {
                    role: 'assistant',
                    content: [
                        {
                            type: 'text',
                            text: assistantMessage
                        }
                    ]
                };
                conversationHistory.push(assistantMessageObj);
            } else {
                throw new Error('API返回格式异常');
            }
        }

        // 滚动到底部
        function scrollToBottom() {
            const chatContainer = document.getElementById('chatContainer');
            chatContainer.scrollTop = chatContainer.scrollHeight;
        }

        // 事件监听
        document.getElementById('userInput').addEventListener('keypress', function(e) {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                sendMessage();
            }
        });

        // 页面加载后聚焦输入框
        window.addEventListener('load', function() {
            document.getElementById('userInput').focus();
        });
    </script>
